\section*{Solution:}
\fi

Given 
\begin{align}
 \vec{0} = \myvec{0\\0},
 \vec{A} = \myvec{-2\\9}
\end{align}
\iffalse
Let D be a point on the required line :
\begin{align}
 \vec{D} = \myvec{x\\y}
\end{align} 
 \begin{align}
 \vec{D-A}=\myvec{x+2\\y-9}
   \end{align}
   \fi
   The 
normal vector is
\begin{align}
	\vec{n}&=\vec{O}- \vec{A}
	\\
	&=\myvec{2\\-9}
\end{align}
yielding the equation of the line as

\begin{align}
	\myvec{2 & -9}\brak{\vec{x} - \myvec{2 \\ -9}}&= 0
	\\
	\implies 
	\myvec{2 & -9}\vec{x} &= 85
\end{align}
\iffalse

\begin{align}
 2\vec{x}-9\vec{y}+85=0
\end{align}
